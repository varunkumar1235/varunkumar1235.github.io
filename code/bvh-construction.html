def build_BVH(objects):
    if len(objects) == 0:
        return None

    # Step 1: Compute the bounding box of all objects
    global_bbox = compute_bounding_box(objects)

    # Step 2: Base case: if few objects, make a leaf node
    if len(objects) <= MAX_LEAF_SIZE:
        return BVHNode(bbox=global_bbox, left=None, right=None, objects=objects)

    # Step 3: Choose the axis to split (longest axis)
    axis = global_bbox.largest_extent_axis()

    # Step 4: Sort objects by the centroid along the chosen axis
    objects.sort(key=lambda obj: obj.centroid()[axis])

    # Step 5: Split the objects into two halves
    mid = len(objects) // 2
    left_objects = objects[:mid]
    right_objects = objects[mid:]

    # Step 6: Recursively build BVH for each half
    left_node = build_BVH(left_objects)
    right_node = build_BVH(right_objects)

    # Step 7: Return parent node
    return BVHNode(bbox=global_bbox, left=left_node, right=right_node)
